                                              通用配置

                                              请参考通用配置

                                              依赖配置

                                              <dependency>
    <groupId>io.github.qifan777</groupId>
    <artifactId>spring-ai-qianfan-spring-boot-starter</artifactId>
</dependency>

                                              密钥配置

                                              千帆 access-key和secret-key申请open in new window

                                              spring:
  ai:
    # 百度千帆
    qian-fan:
      access-key: xxx
      secret-key: xxx
      enabled: true
      chat:
        model: ERNIE-4.0-8K

                                              private final QianFanAiChatModel qianFanAiChatModel;
