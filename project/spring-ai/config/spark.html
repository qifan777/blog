                          讯飞星火接入

                          通用配置

                          请参考通用配置

                          依赖配置

                          <dependency>
    <groupId>io.github.qifan777</groupId>
    <artifactId>spring-ai-spark-spring-boot-starter</artifactId>
</dependency>

                          密钥配置

                          api-key、api-secret、appid申请地址open in new window

                          spring:
  ai:
    # 讯飞星火
    spark:
      chat:
        api-key: xxx
        api-secret: xxx
        appid: xxx
        enabled: true
        options:
          # 设置模型的domain
          domain: generalv3.5
          # 设置模型的baseUrl
          base-url: https://spark-api.xf-yun.com/v3.5/chat

                          模型使用

                          private final SparkAiChatModel sparkAiChatModel;
